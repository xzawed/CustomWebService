## 테이블 정의

### 1. users (사용자)
```sql
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email VARCHAR(255) UNIQUE NOT NULL,
    name VARCHAR(100),
    avatar_url TEXT,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);
```
- Supabase Auth와 연동, 소셜 로그인 지원

### 2. api_catalog (API 카탈로그)
```sql
CREATE TABLE api_catalog (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(200) NOT NULL,
    description TEXT NOT NULL,
    category VARCHAR(50) NOT NULL,
    base_url VARCHAR(500) NOT NULL,
    auth_type VARCHAR(20) NOT NULL DEFAULT 'none',  -- none, api_key, oauth
    auth_config JSONB DEFAULT '{}',                  -- 인증 관련 설정
    rate_limit VARCHAR(100),                         -- "1000/day" 형식
    is_active BOOLEAN DEFAULT true,
    icon_url VARCHAR(500),
    docs_url VARCHAR(500),
    endpoints JSONB NOT NULL DEFAULT '[]',           -- 사용 가능 엔드포인트 목록
    tags TEXT[] DEFAULT '{}',
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 카테고리별 조회를 위한 인덱스
CREATE INDEX idx_api_catalog_category ON api_catalog(category);
CREATE INDEX idx_api_catalog_is_active ON api_catalog(is_active);
```

#### endpoints JSONB 구조 예시
```json
[
    {
        "path": "/weather",
        "method": "GET",
        "description": "현재 날씨 조회",
        "params": [
            {"name": "q", "type": "string", "required": true, "description": "도시명"},
            {"name": "units", "type": "string", "required": false, "description": "단위 (metric/imperial)"}
        ],
        "response_example": {"temp": 20, "description": "맑음"}
    }
]
```

### 3. projects (프로젝트 / 생성된 서비스)
```sql
CREATE TABLE projects (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    name VARCHAR(200) NOT NULL,
    context TEXT NOT NULL,                           -- 사용자가 입력한 서비스 설명
    status VARCHAR(20) DEFAULT 'draft',              -- draft, generating, generated, deploying, deployed, failed
    deploy_url VARCHAR(500),                         -- 배포된 서비스 URL
    deploy_platform VARCHAR(20),                     -- vercel, netlify, github_pages
    repo_url VARCHAR(500),                           -- GitHub 저장소 URL
    preview_url VARCHAR(500),                        -- 미리보기 URL
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_projects_user_id ON projects(user_id);
CREATE INDEX idx_projects_status ON projects(status);
```

### 4. project_apis (프로젝트-API 매핑)
```sql
CREATE TABLE project_apis (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
    api_id UUID NOT NULL REFERENCES api_catalog(id),
    config JSONB DEFAULT '{}',                       -- API별 설정 (API 키 등)
    created_at TIMESTAMPTZ DEFAULT NOW(),
    UNIQUE(project_id, api_id)
);

CREATE INDEX idx_project_apis_project_id ON project_apis(project_id);
```

### 5. generated_codes (생성된 코드)
```sql
CREATE TABLE generated_codes (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
    version INTEGER NOT NULL DEFAULT 1,
    code_html TEXT,
    code_css TEXT,
    code_js TEXT,
    framework VARCHAR(20) DEFAULT 'vanilla',          -- vanilla, react, next
    ai_prompt_used TEXT,                              -- 사용된 AI 프롬프트 (디버깅용)
    created_at TIMESTAMPTZ DEFAULT NOW(),
    UNIQUE(project_id, version)
);

CREATE INDEX idx_generated_codes_project_id ON generated_codes(project_id);
```

## RLS (Row Level Security) 정책

```sql
-- 사용자는 자신의 데이터만 접근
ALTER TABLE projects ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can CRUD own projects" ON projects
    FOR ALL USING (auth.uid() = user_id);

ALTER TABLE project_apis ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can CRUD own project_apis" ON project_apis
    FOR ALL USING (
        project_id IN (SELECT id FROM projects WHERE user_id = auth.uid())
    );

ALTER TABLE generated_codes ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can read own generated_codes" ON generated_codes
    FOR SELECT USING (
        project_id IN (SELECT id FROM projects WHERE user_id = auth.uid())
    );

-- API 카탈로그는 모든 인증된 사용자 읽기 가능
ALTER TABLE api_catalog ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Authenticated users can read api_catalog" ON api_catalog
    FOR SELECT USING (auth.role() = 'authenticated');
```
